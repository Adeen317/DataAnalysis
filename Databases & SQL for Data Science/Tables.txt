#Table For ChicagoCrimeData

CREATE TABLE Chicago_Crime_Data (
    ID INT NOT NULL AUTO_INCREMENT,
    CASE_NUMBER VARCHAR(25) NOT NULL,
    Date_Reg DATE NOT NULL,
    BLOCK_NUMBER VARCHAR(50) NOT NULL,
    IUCR INT(25) NOT NULL,
    PRIMARY_TYPE VARCHAR(20) NOT NULL,
    Description_crime VARCHAR(25) NOT NULL,
    LOCATION_DESCRIPTION VARCHAR(25) NOT NULL,
    ARREST VARCHAR(6) NOT NULL,
    DOMESTIC VARCHAR(6) NOT NULL,
    BEAT INT(10),
    DISTRICT INT(10),
    WARD INT(10),
    COMMUNITY_AREA_NUMBER INT(10),
    FBI_CODE INT(10),
    X_CORDINATE INT(15) not NUll,
    Y_CORDINATE INT(15) not NUll,
    Year_Crime INT(4) not NUll,
    LATITUDE INT(20) not NUll,
    LONGITUDE INT(20) not NUll,
    LOCATION VARCHAR(30) not NUll,
    
    PRIMARY KEY (ID)
);
